forM_ = flip mapM_

import Control.Monad

main = do
    forM_ [1..3] $ \i -> do
        print i

    forM_ [7..9] $ \j -> do
        print j

    forM_ [1..] $ \_ -> do
        putStrLn "loop"
        error "break"
